56 function fixPNG(myImage)
57 {
58     if ((version >= 5.5) && (version < 7) && (document.body.filters))
59     {
60        var imgID = (myImage.id) ? "id='" + myImage.id + "' " : ""
61        var imgClass = (myImage.className) ? "class='" + myImage.className + "' " : ""
62        var imgTitle = (myImage.title) ?
63                      "title='" + myImage.title  + "' " : "title='" + myImage.alt + "' "
64        var imgStyle = "display:inline-block;" + myImage.style.cssText
65        var strNewHTML = "<span " + imgID + imgClass + imgTitle
66                   + " style=\"" + "width:" + myImage.width
67                   + "px; height:" + myImage.height
68                   + "px;" + imgStyle + ";"
69                   + "filter:progid:DXImageTransform.Microsoft.AlphaImageLoader"
70                   + "(src=\'" + myImage.src + "\', sizingMethod='scale');\"></span>"
71        myImage.outerHTML = strNewHTML
72     }
73 }
